def randsp(n, density):
    nnz = int(n*n*density)
    row = np.random.random_integers(low=0, high=n-1, size=nnz)
    col = np.random.random_integers(low=0, high=n-1, size=nnz)
    data = np.ones(nnz, dtype=float)

    A = sparse.coo_matrix((data, (row, col)), shape=(n, n))
    return A
