def load_image(name):
    with Image.open(name).resize((500, 500)) as img: #this opens the image file and resizes it to size 500x500x3 (row, column, Red/Blue/Green)
        return np.sum(np.array(img), axis=-1).astype(np.float32)/(3*255) #converts image to numpy array and sums along 3rd axis to convert to grayscale
